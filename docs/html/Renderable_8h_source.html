163bool Renderable::DepthOrder< order >::operator()( const Renderable& l, const Renderable& r ) const

169template< unsigned int role >
170bool Renderable::VideoResourcesOrder< role >::operator()( const Renderable& l, const Renderable& r ) const
171{
172 const GeometryFeature& gfl = l.geometry().feature( role );
173 const GeometryFeature& gfr = r.geometry().feature( role );
174 if( gfl.controlsSameVideoResource( gfr ) )
175 {
176 return false;
177 }
178 else
179 {
180 return &gfl < &gfr;

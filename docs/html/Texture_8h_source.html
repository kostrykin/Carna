272template< unsigned int dimension >
273void Texture< dimension >::update( const Eigen::Matrix< unsigned int, dimension, 1 >& size, int bufferType, const void* bufferPtr )
274{
275 /* Ensure that texture size is positive.
276 */
277 for( unsigned int i = 0; i < dimension; ++i )
278 {
279 CARNA_ASSERT_EX( size( i, 0 ) >= 1, "Texture only supports positive sizes!" );
280 }
281
282 /* Ensure that z-component of the texture size is even if this is a 3D texture.
283 */
284 if( dimension == 3 )
285 {
286 CARNA_ASSERT_EX( size( 2, 0 ) % 2 == 0, "3D textures must have even depth!" );
287 }
288
289 /* Update the OpenGL texture object.
290 */
291 this->bind( SETUP_UNIT );
292 uploadGLTextureData( size, internalFormat, pixelFormat, bufferType, bufferPtr );
293}

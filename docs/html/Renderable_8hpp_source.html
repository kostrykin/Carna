165template< int order >
166bool Renderable::DepthOrder< order >::operator()( const Renderable& l, const Renderable& r ) const
167{
168 return order * ( l.eyeDistance2() - r.eyeDistance2() ) > 0;
169}

172template< unsigned int role >
173bool Renderable::VideoResourcesOrder< role >::operator()( const Renderable& l, const Renderable& r ) const
174{
175 const GeometryFeature& gfl = l.geometry().feature( role );
176 const GeometryFeature& gfr = r.geometry().feature( role );
177 if( gfl.controlsSameVideoResource( gfr ) )
178 {
179 return false;
180 }
181 else
182 {
183 return &gfl < &gfr;
184 }

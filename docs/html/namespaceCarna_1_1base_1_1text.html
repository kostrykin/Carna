Performs lexical cast.

Example:

using Carna::base::text::lexical_cast;
const unsigned int i = 3;
const std::string s = lexical_cast< std::string >( i );
const unsigned int j = lexical_cast< unsigned int >( s );
CARNA_ASSERT( i == j );

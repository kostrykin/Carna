124template< typename VertexType >
125ManagedMesh< VertexType, uint8_t >& MeshFactory< VertexType >::createBox( float sizeX, float sizeY, float sizeZ )
126{
127 const math::Matrix4f baseTransform = math::scaling4f( sizeX / 2, sizeY / 2, sizeZ / 2 );
128
129 /* Define faces.
130 */
131 math::Matrix4f transforms[ 6 ];
132 transforms[ 0 ] = math::basis4f( math::Vector3f( 0, 0, +1 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( -1, 0, 0 ) ); // left
133 transforms[ 1 ] = math::basis4f( math::Vector3f( 0, 0, -1 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( +1, 0, 0 ) ); // right
134 transforms[ 2 ] = math::basis4f( math::Vector3f( +1, 0, 0 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( 0, 0, +1 ) ); // front
135 transforms[ 3 ] = math::basis4f( math::Vector3f( -1, 0, 0 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( 0, 0, -1 ) ); // back
136 transforms[ 4 ] = math::basis4f( math::Vector3f( +1, 0, 0 ), math::Vector3f( 0, 0, -1 ), math::Vector3f( 0, +1, 0 ) ); // top
137 transforms[ 5 ] = math::basis4f( math::Vector3f( +1, 0, 0 ), math::Vector3f( 0, 0, +1 ), math::Vector3f( 0, -1, 0 ) ); // bottom
138
139 const std::size_t verticesCount = 6 * 4;
140 const std::size_t indicesCount = 6 * 2 * 3;
141
142 typedef ManagedMesh< VertexType, uint8_t > MeshInstance;
143 typedef typename MeshInstance::Vertex Vertex;
144 typedef typename MeshInstance:: Index Index;
145 Vertex vertices[ verticesCount ];
146 Index indices[ indicesCount ];
147
148 int lastVertex = -1;
149 int lastIndex = -1;
150
151 /* Create vertices and indices.
152 */
153 for( unsigned int faceIndex = 0; faceIndex < 6; ++faceIndex )
154 {
155 const auto positionTransform = baseTransform * transforms[ faceIndex ];
156 const auto normalTransform = positionTransform.inverse().transpose();
157 const auto normalVector = ( normalTransform * math::Vector4f( 0, 0, 1, 0 ) ).normalized();
158
159 vertices[ ++lastVertex ] = vertex< math::Vector4f >( positionTransform * math::Vector4f( -1, -1, 1, 1 ), normalVector );
160 vertices[ ++lastVertex ] = vertex< math::Vector4f >( positionTransform * math::Vector4f( +1, -1, 1, 1 ), normalVector );
161 vertices[ ++lastVertex ] = vertex< math::Vector4f >( positionTransform * math::Vector4f( +1, +1, 1, 1 ), normalVector );
162 vertices[ ++lastVertex ] = vertex< math::Vector4f >( positionTransform * math::Vector4f( -1, +1, 1, 1 ), normalVector );
163
164 indices[ ++lastIndex ] = lastVertex - 3;
165 indices[ ++lastIndex ] = lastVertex - 2;
166 indices[ ++lastIndex ] = lastVertex;
167
168 indices[ ++lastIndex ] = lastVertex;
169 indices[ ++lastIndex ] = lastVertex - 2;
170 indices[ ++lastIndex ] = lastVertex - 1;
171 }
172
173 return MeshInstance::create
174 ( IndexBufferBase::PRIMITIVE_TYPE_TRIANGLES
175 , vertices, verticesCount
176 , indices, indicesCount );
177}

180template< typename VertexType >
181ManagedMesh< VertexType, uint16_t >& MeshFactory< VertexType >::createBall( float radius, unsigned int degree )
182{
183 const math::Matrix4f baseTransform = math::scaling4f( radius, radius, radius );
184 const unsigned int verticesPerEdge = 2 + degree;
185 const unsigned int verticesPerSide = verticesPerEdge * verticesPerEdge;
186 const unsigned int facesPerSide = ( verticesPerEdge - 1 ) * ( verticesPerEdge - 1 );
187 const unsigned int indicesPerSide = 6 * facesPerSide;
188
189 /* Define sides.
190 */
191 math::Matrix4f transforms[ 6 ];
192 transforms[ 0 ] = math::basis4f( math::Vector3f( 0, 0, +1 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( -1, 0, 0 ) ); // left
193 transforms[ 1 ] = math::basis4f( math::Vector3f( 0, 0, -1 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( +1, 0, 0 ) ); // right
194 transforms[ 2 ] = math::basis4f( math::Vector3f( +1, 0, 0 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( 0, 0, +1 ) ); // front
195 transforms[ 3 ] = math::basis4f( math::Vector3f( -1, 0, 0 ), math::Vector3f( 0, +1, 0 ), math::Vector3f( 0, 0, -1 ) ); // back
196 transforms[ 4 ] = math::basis4f( math::Vector3f( +1, 0, 0 ), math::Vector3f( 0, 0, -1 ), math::Vector3f( 0, +1, 0 ) ); // top
197 transforms[ 5 ] = math::basis4f( math::Vector3f( +1, 0, 0 ), math::Vector3f( 0, 0, +1 ), math::Vector3f( 0, -1, 0 ) ); // bottom
198
199 const std::size_t verticesCount = 6 * verticesPerSide;
200 const std::size_t indicesCount = 6 * indicesPerSide;
201 LIBCARNA_ASSERT(indicesCount < ( 1 << 16 ));
202
203 typedef ManagedMesh< VertexType, uint16_t > MeshInstance;
204 typedef typename MeshInstance::Vertex Vertex;
205 typedef typename MeshInstance:: Index Index;
206 Vertex vertices[ verticesCount ];
207 Index indices[ indicesCount ];
208
209 int lastVertex = -1;
210 int lastIndex = -1;
211
212 /* Create vertices and indices.
213 */
214 for( unsigned int sideIndex = 0; sideIndex < 6; ++sideIndex )
215 {
216 const auto positionTransform = baseTransform * transforms[ sideIndex ];
217 const auto normalTransform = positionTransform.inverse().transpose();
218
219 for( unsigned int y = 0; y < verticesPerEdge; ++y )
220 for( unsigned int x = 0; x < verticesPerEdge; ++x )
221 {
222 const float fx = 2 * x / float( verticesPerEdge - 1 );
223 const float fy = 2 * y / float( verticesPerEdge - 1 );
224 const auto position = math::Vector3f( -1 + fx, -1 + fy, 1 ).normalized();
225 vertices[ ++lastVertex ] = vertex< math::Vector4f >
226 ( positionTransform * math::Vector4f( position.x(), position.y(), position.z(), 1 )
227 , normalTransform * math::Vector4f( position.x(), position.y(), position.z(), 0 )
228 );
229 }
230
231 for( unsigned int y = 0; y < verticesPerEdge - 1; ++y )
232 for( unsigned int x = 0; x < verticesPerEdge - 1; ++x )
233 {
234 const uint16_t ul = x + y * verticesPerEdge; // upper left
235 const uint16_t ur = x + 1 + y * verticesPerEdge; // upper right
236 const uint16_t ll = x + ( y + 1 ) * verticesPerEdge; // lower left
237 const uint16_t lr = x + 1 + ( y + 1 ) * verticesPerEdge; // lower right
238
239 indices[ ++lastIndex ] = verticesPerSide * sideIndex + ul;
240 indices[ ++lastIndex ] = verticesPerSide * sideIndex + lr;
241 indices[ ++lastIndex ] = verticesPerSide * sideIndex + ll;
242
243 indices[ ++lastIndex ] = verticesPerSide * sideIndex + ul;
244 indices[ ++lastIndex ] = verticesPerSide * sideIndex + ur;
245 indices[ ++lastIndex ] = verticesPerSide * sideIndex + lr;
246 }
247 }
248
249 return MeshInstance::create
250 ( IndexBufferBase::PRIMITIVE_TYPE_TRIANGLES
251 , vertices, verticesCount
252 , indices, indicesCount );
253}

275template< typename VertexType >
276ManagedMesh< VertexType, uint32_t >& MeshFactory< VertexType >::createFromSTL( std::istream& stlStream )
277{
278 auto origExceptMask = stlStream.exceptions();
279 stlStream.exceptions(std::istream::failbit | std::istream::badbit | std::istream::eofbit);
280
281 uint32_t amountTriangles;
282
283 //discard first 80 bytes
284 stlStream.seekg(stlStream.beg + 80);
285
286 //read amount of triangles (32bit-uint)
287 stlStream.read(reinterpret_cast<char*>(&amountTriangles), 4);
288
289 typedef ManagedMesh< VertexType, uint32_t > MeshInstance;
290 typedef typename MeshInstance::Vertex Vertex;
291 typedef typename MeshInstance::Index Index;
292 typedef typename std::pair< Vertex, std::size_t > VertexIndexPair;
293
294 std::size_t indicesCount = amountTriangles * 3;
295 std::unique_ptr<Index[]> indices (new Index [indicesCount]);
296 std::unique_ptr<Vertex[]> vertices(new Vertex [indicesCount]);
297
298 std::function<std::size_t(const VertexIndexPair&)> vertexHash =
299 [](const VertexIndexPair& vertAndIndex) -> std::size_t
300 {
301 //creates combined hash value of the 3 public members "x","y","z" of a Vertex object; ignores the second member of pair
302 //This hash-combine-technique is taken from : http://www.open-std.org/jtc1/sc22/wg21/docs/papers/2014/n3876.pdf (Page 2)
303 std::size_t hashVal = std::hash<decltype(vertAndIndex.first.x)>()(vertAndIndex.first.x);
304 hashVal ^= std::hash<decltype(vertAndIndex.first.y)>()(vertAndIndex.first.y) + 0x9e3779b9 + (hashVal << 6) + (hashVal >> 2);
305 hashVal ^= std::hash<decltype(vertAndIndex.first.z)> ()(vertAndIndex.first.z) + 0x9e3779b9 + (hashVal << 6) + (hashVal >> 2);
306 return hashVal;
307 };
308
309 std::function<std::size_t(const VertexIndexPair&, const VertexIndexPair&)> vertexComp =
310 [](const VertexIndexPair& vertAndIndex1, const VertexIndexPair& vertAndIndex2) -> bool
311 {
312 //checks whether the members "x", "y", "z" of two Vertex objects are equal
313 return ((vertAndIndex1.first.x) == (vertAndIndex2.first.x)) && ((vertAndIndex1.first.y) == (vertAndIndex2.first.y)) && ((vertAndIndex1.first.z) == (vertAndIndex2.first.z));
314 };
315
316 std::unordered_set< VertexIndexPair, decltype(vertexHash), decltype(vertexComp) > vertsWithArrIndices(amountTriangles * 3, vertexHash, vertexComp);
317
318 std::size_t indicesItCount = 0;
319 std::size_t verticesItCount = 0;
320
321 //read triangles
322 for (unsigned int i = 0; i < amountTriangles; i++)
323 {
324 //discard normal
325 stlStream.seekg(12, std::ios_base::cur);
326
327 //read vertices
328 for (unsigned int j = 0; j < 3; j++) //read 3 vertices
329 {
330 float x, y, z;
331
332 //read 1 vertex
333 stlStream.read(reinterpret_cast<char*>(&x), 4);
334 stlStream.read(reinterpret_cast<char*>(&y), 4);
335 stlStream.read(reinterpret_cast<char*>(&z), 4);
336
337 Vertex vert;
338 vert.x = x;
339 vert.y = y;
340 vert.z = z;
341 //vert gets invalidated here
342 auto emplRes = vertsWithArrIndices.emplace(std::piecewise_construct, std::forward_as_tuple(std::move(vert)), std::forward_as_tuple(verticesItCount));
343
344 if (emplRes.second)
345 {
346 vertices[verticesItCount++] = (emplRes.first)->first;
347 }
348
349 indices[indicesItCount++] = (emplRes.first)->second;
350 }
351
352 //discard 2 bytes
353 stlStream.seekg(2, std::ios_base::cur);
354 }
355
356 stlStream.exceptions(origExceptMask);
357
358 return MeshInstance::create
359 (IndexBufferBase::PRIMITIVE_TYPE_TRIANGLES
360 , vertices.get(), verticesItCount
361 , indices.get(), indicesCount);
362}

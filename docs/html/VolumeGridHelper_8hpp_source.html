357template< typename SegmentIntensityVolumeType, typename SegmentNormalsVolumeType >
358base::math::Vector3ui VolumeGridHelper< SegmentIntensityVolumeType, SegmentNormalsVolumeType >::computeMaxSegmentSize
359 ( const base::math::Vector3ui& nativeResolution, std::size_t maxSegmentBytesize )
360{
361 const float maxSideLengthF = std::pow
362 ( maxSegmentBytesize / static_cast< float >( sizeof( typename SegmentIntensityVolumeType::Voxel ) ), 1.f / 3 );
363 const unsigned int maxSideLength = base::math::makeEven( base::math::round_ui( maxSideLengthF ), -1 );
364
365 /* We subtract the redundant texels from effective segment size.
366 * Note that this causes the effective maximum segment size to be odd.
367 */
368 return base::math::Vector3ui( maxSideLength - 1, maxSideLength - 1, maxSideLength - 1 );
369}

372template< typename SegmentIntensityVolumeType, typename SegmentNormalsVolumeType >
373VolumeGridHelper< SegmentIntensityVolumeType, SegmentNormalsVolumeType >::VolumeGridHelper
374 ( const base::math::Vector3ui& nativeResolution
375 , std::size_t maxSegmentBytesize )
376 : VolumeGridHelperBase( nativeResolution )
377 , maxSegmentBytesize( maxSegmentBytesize )
378 , maxSegmentSize( computeMaxSegmentSize( nativeResolution, maxSegmentBytesize ) )
379 , partitioningX( nativeResolution.x(), maxSegmentSize.x() )
380 , partitioningY( nativeResolution.y(), maxSegmentSize.y() )
381 , partitioningZ( nativeResolution.z(), maxSegmentSize.z() )
382 , resolution( partitioningX.totalSize(), partitioningY.totalSize(), partitioningZ.totalSize() )
383{
384 const base::math::Vector3ui segmentCounts
385 ( partitioningX.partitionsCount()
386 , partitioningY.partitionsCount()
387 , partitioningZ.partitionsCount() );
388 myGrid.reset( new base::VolumeGrid< SegmentIntensityVolumeType, SegmentNormalsVolumeType >( maxSegmentSize, segmentCounts ) );
389 NormalsComponent::setGrid( *myGrid );
390
391 LIBCARNA_FOR_VECTOR3UI( segmentCoord, myGrid->segmentCounts )
392 {
393 /* Here we add the redundant texels to the buffer size considerations.
394 */
395 const base::math::Vector3ui segmentSize
396 ( segmentCoord.x() + 1 == myGrid->segmentCounts.x() ? partitioningX.tailSize : partitioningX.regularPartitionSize + 1
397 , segmentCoord.y() + 1 == myGrid->segmentCounts.y() ? partitioningY.tailSize : partitioningY.regularPartitionSize + 1
398 , segmentCoord.z() + 1 == myGrid->segmentCounts.z() ? partitioningZ.tailSize : partitioningZ.regularPartitionSize + 1 );
399
400 IntensityComponent::initializeSegment( myGrid->segmentAt( segmentCoord ), segmentSize );
401 NormalsComponent ::initializeSegment( myGrid->segmentAt( segmentCoord ), segmentSize );
402 }
403}

440template< typename SegmentIntensityVolumeType, typename SegmentNormalsVolumeType >
441base::Node* VolumeGridHelper< SegmentIntensityVolumeType, SegmentNormalsVolumeType >::createNode
442 ( unsigned int geometryType, const Spacing& spacing, const Extent& extent ) const
443{
444 /* Compute the extent of a regular grid segment, taking the redundant texels into account. Regular segments have
445 * `regularPartitionSize + 1` texels per dimension which is due to the redundant pixels.
446 */
447 const base::math::Vector3f regularSegmentExtent = extent.units.cwiseMin( base::math::Vector3f
448 ( spacing.units.x() * partitioningX.regularPartitionSize
449 , spacing.units.y() * partitioningY.regularPartitionSize
450 , spacing.units.z() * partitioningZ.regularPartitionSize ) );
451
452 /* Create pivot node that centers its children.
453 *
454 * Suppose the following example grid layout, where p is the absolute center and
455 * q is the center of the first segment:
456 *
457 * o-------o-------o---o
458 * | q | | |
459 * o-------o-------o---o
460 * | | p | |
461 * o-------o-------o---o
462 * | | | |
463 * o-------o-------o---o
464 *
465 * Each segment is rendered around its local center. Thus, the translation T
466 * required to center the volume geometry in the origin, corresponds to:
467 *
468 * T = -p + q
469 */
470 base::Node* const pivot = new base::Node();
471 pivot->localTransform = base::math::translation4f( ( regularSegmentExtent - extent.units ) / 2 );
472 pivot->setMovable( false );
473
474 /* Create geometry nodes for all grid segments.
475 */
476 LIBCARNA_FOR_VECTOR3UI( segmentCoord, myGrid->segmentCounts )
477 {
478 const base::VolumeSegment< SegmentIntensityVolumeType, SegmentNormalsVolumeType >& segment = myGrid->segmentAt( segmentCoord );
479
480 /* Compute extent of particular grid segment.
481 */
482 const bool isTailX = segmentCoord.x() + 1 == myGrid->segmentCounts.x();
483 const bool isTailY = segmentCoord.y() + 1 == myGrid->segmentCounts.y();
484 const bool isTailZ = segmentCoord.z() + 1 == myGrid->segmentCounts.z();
485 const base::math::Vector3ui& volumeSize = segment.intensities().size; // includes redundant pixels (only along non-tail axes)
486 const base::math::Vector3f segmentExtent
487 ( isTailX ? ( volumeSize.x() - 1 ) * spacing.units.x() : regularSegmentExtent.x()
488 , isTailY ? ( volumeSize.y() - 1 ) * spacing.units.y() : regularSegmentExtent.y()
489 , isTailZ ? ( volumeSize.z() - 1 ) * spacing.units.z() : regularSegmentExtent.z() );
490
491 /* Create geometry node for particular grid segment.
492 */
493 base::Geometry* const geom = new base::Geometry( geometryType );
494 pivot->attachChild( geom );
495 IntensityComponent::attachTexture( *geom, segment );
496 NormalsComponent ::attachTexture( *geom, segment );
497 geom->setMovable( false );
498 geom->setBoundingVolume( new base::BoundingBox( 1, 1, 1 ) );
499 geom->localTransform
500 = base::math::translation4f
501 ( segmentCoord.x() * regularSegmentExtent.x() - ( isTailX ? ( regularSegmentExtent.x() - segmentExtent.x() ) / 2 : 0 )
502 , segmentCoord.y() * regularSegmentExtent.y() - ( isTailY ? ( regularSegmentExtent.y() - segmentExtent.y() ) / 2 : 0 )
503 , segmentCoord.z() * regularSegmentExtent.z() - ( isTailZ ? ( regularSegmentExtent.z() - segmentExtent.z() ) / 2 : 0 ) )
504 * base::math::scaling4f( segmentExtent );
505 }
506
507 /* We're done.
508 */
509 base::Log::instance().record( base::Log::debug
510 , "VolumeGridHelper computed "
511 + base::text::lexical_cast< std::string >( 8 * sizeof( typename SegmentIntensityVolumeType::Voxel ) )
512 + "bit grid data using "
513 + base::text::lexical_cast< std::string >( myGrid->segmentCounts.x() ) + ""
514 + base::text::lexical_cast< std::string >( myGrid->segmentCounts.y() ) + ""
515 + base::text::lexical_cast< std::string >( myGrid->segmentCounts.z() ) + " segments" );
516 return pivot;
517}
